git init - инициализируем репозиторий один раз

git status - проверка статуса
git add - добавление в индексированную область


git log - просмотр всех коммитов
git log --oneline


git commit - коммит - сохраниение снимка
-m - позволяет добавить название без открытия редактора
--amend - позволяет внести изменения(отменить) в последний коммит
